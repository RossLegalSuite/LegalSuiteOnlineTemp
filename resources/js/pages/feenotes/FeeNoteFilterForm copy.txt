
<template>
    <div class="modal" :id="id">
        <div class="modal-dialog modal-dialog-centered modal-lg">
            <div class="modal-content" style="border-color: indianred;">

            <div class="modal-header indianred">
                <h3 class="modal-title"><i class="fa fa-filter mr-2"></i>Filter Fee Notes</h3>
                <i title="Close this window" class="cp top-right fa fa-times-circle fa-2x text-white" data-dismiss="modal"></i>
            </div>

            <div class="modal-body p-3" style="height: 50vh; overflow-y: auto;">

                <div class="form-group row">
                    <div class="col-md-12">
                        <b-form-checkbox
                            class="mt-1 mb-1"
                            title="Display Fee Notes you created"
                            v-model="myRecordsSelected"
                            value="1"
                            unchecked-value="0"
                            @change="myRecordsChanged"
                        >
                        Display My Fee Notes
                        </b-form-checkbox>

                    </div>
                </div>

                <div class="form-group row">

                    <div class="col-md-6">

                        <span title="Display Fee Notes that have not been Invoiced">

                            <b-form-checkbox
                                class="mt-1"
                                v-model="notInvoicedSelected"
                                value="1"
                                unchecked-value="0"
                                @change="notInvoicedChanged"
                            >
                            Display Unposted Fee Notes
                            </b-form-checkbox>

                        </span>

                    </div>

                </div>

                <div class="form-group row">

                    <div class="col-md-6">

                        <span title="Display Fee Notes that have been Invoiced">

                            <b-form-checkbox
                                class="mt-1"
                                v-model="invoicedSelected"
                                value="1"
                                unchecked-value="0"
                                @change="invoicedChanged"
                            >
                            Display Invoiced Fee Notes
                            </b-form-checkbox>

                        </span>

                    </div>

                </div>

                <select-date-filter
                    :id="id + '-date-filter'"
                    :ref="id + '-date-filter'"
                />

                <amount-filter
                    :id="id + '-amount-filter'"
                    :ref="id + '-amount-filter'"
                />

            </div>

            <div class="modal-footer justify-content-between">
                <div>
                    <button class="btn btn-primary form-button mr-2" type="button" @click="loadMoreFilters" title="Advanced Filtering options"><i class="fa fa-cog fa-lg mr-2"></i>Advanced</button>
                </div>
                <div>
                    <button class="btn btn-danger form-button" type="button" @click="hide" title="Close this screen"><i class="fa fa-times-circle fa-lg mr-2"></i>Close</button>
                </div>
            </div>

            </div>
        </div>
    </div>
</template>

<script>

import modalTemplate from "@components/modals/modal-template";
import filterMethods from "@components/filters/filter-methods";

export default {

    mixins: [modalTemplate, filterMethods],

    data() {
        return {
            myRecordsSelected: '0',
            myRecordsFilterColumnName: 'created.name',
            dateFilterColumnName: 'date',
            invoicedSelected: '0',
            notInvoicedSelected: '0',
            amountFilterColumnName: 'fee_notes.totalAmount',
        }
    },

    mounted() {
        this.$parent.filterForm = this;
    },    

    methods: {

        modalOpened() {

            // Prime the myRecords Filter Check Box
            this.myRecordsFilterKey = this.getFilterName(this.myRecordsFilterColumnName);
            this.myRecordsFilter = this.getFilter(this.myRecordsFilterKey);
            this.myRecordsSelected = this.myRecordsFilter ? '1' : '0';

            // Prime the Date Filter 
            let dateFilterKey = this.getFilterName(this.dateFilterColumnName);
            this.$refs[this.id + '-date-filter'].primeFilter(dateFilterKey, this.dateFilterColumnName);

            // Prime the Amount Filter component
            let amountFilterKey = this.getFilterName(this.amountFilterColumnName);
            this.$refs[this.id + '-amount-filter'].primeFilter(amountFilterKey, this.amountFilterColumnName);

            // Prime the Check Boxes
            let invoicedFilter = this.getFilter('FeeNotesInvoicedFilter');
            this.invoicedSelected = invoicedFilter ? '1' : '0';

            let notInvoicedFilter = this.getFilter('FeeNotesNotInvoicedFilter');
            this.notInvoicedSelected = notInvoicedFilter ? '1' : '0';



        },

        invoicedChanged(newValue) {

            this.deleteFilter('FeeNotesInvoicedFilter');

            if (newValue == 1) this.addFilter('invoicePosted', 'FeeNotesInvoicedFilter', '1');

            this.$parent.setUserFilters();

            this.$parent.table.ajax.reload();
            
        },

        notInvoicedChanged(newValue) {

            this.deleteFilter('FeeNotesNotInvoicedFilter');

            if (newValue == 1) this.addFilter('invoicePosted', 'FeeNotesNotInvoicedFilter', '0');

            this.$parent.setUserFilters();

            this.$parent.table.ajax.reload();
            
        },


    }   
}  
</script>
