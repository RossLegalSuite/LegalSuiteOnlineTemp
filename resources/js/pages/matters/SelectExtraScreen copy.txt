<template>
<div :class="formControls ? _class : ''">

    <div v-show="formControls">
        
        <label>{{label}}</label>
        <span v-if="popOver">   
            <pop-over :content="popOver" :container="popOverContainer ? popOverContainer : $parent.modal ? '#' + $parent.id : 'body'"/>
        </span>
        <span v-if="clearIcon">
            <i class="fa fa-times-circle cp text-danger ml-1 clear-icon" style="font-size: large" title="Clear this filter" @click="clearIconClicked"></i>
        </span>
        <div class="input-group">
            <select-extra-screen-dropdown
                :id="id + '-select-extra-screen-dropdown'"
                :class="error[0] ? 'multiselect-invalid-feedback' : ''"
                openDirection="below"
                placeholder=""
                :options="setDropDownOptions()" 
                v-model="record"
                track-by="recordid"
                :tabindex="-1"
                label="description"
                :disabled="readOnly || disabled"
                :allow-empty="false"
                :show-labels="false"
                @input="extraScreenSelected"
                :loading="setLoadingIndicator()"
                :internal-search="false"
                @search-change="debounceFindExtraScreens"
            >
            <template slot="noResult">No Extra Screens found</template>
            <template slot="noOptions">No Extra Screens found...</template>
            <template slot="option" slot-scope="props">
                <div>
                    <span>{{ props.option.description }}</span>
                </div>
            </template>                            
            </select-extra-screen-dropdown>

            <div v-if="!readOnly" class="input-group-append">
                <button
                    tabindex="-1"
                    v-on:click="selectExtraScreen"
                    type="button"
                    :title="'Select a ' + singular"
                    :class="error[0] ? 'btn btn-danger multiselect-invalid-feedback' : 'btn btn-primary'"
                ><i class="fa fa-cog"></i></button>
            </div>
        </div>
        <div v-html="error[0]" class="invalid-feedback"></div>


    </div>

    <div class="modal" :id="id">

        <div class="modal-dialog modal-dialog-left">

            <div class="modal-content" style="border-color:indianred">

                <div class="modal-header indianred">
                    <h2 class="modal-title page-title"><i class="fa fa-server mr-2"></i>Select an {{singular}}</h2>
                    <i title="Close this window" class="cp top-right fa fa-times-circle fa-2x text-white" data-dismiss="modal"></i>
                </div>

                <div class="modal-body p-3" style="height: 60vh; overflow-y: auto;">
                    
                    <extra-screen-table
                    :options-button="false"
                    :table-id="id + '-table'"
                    :lazyLoadFlag="true"
                    ref="select-extra-screen-table"
                    :formRef="formRef"
                    />


                </div>

                <div class="modal-footer ">
                    <button class="btn btn-danger form-button" type="button"  v-on:click="hide" title="Cancel selection"><i class="fa fa-times-circle fa-lg mr-2"></i>Cancel</button>
                </div>

            </div>

        </div>

    </div>

</div>
</template>

<script>

import modalTemplate from "@components/modals/modal-template";
import { debounce } from 'lodash';
import Multiselect from 'vue-multiselect';

export default {

    mixins: [modalTemplate],       

    components: {
        ExtraScreenTable: () => import("./ExtraScreenTable"),
        SelectExtraScreenDropdown: Multiselect
    },

    props: {
        clearIconCallback: { type: String, default: 'clearSelectExtraScreen' },
        clearIcon: { type: Boolean, default: false },
        title: {
            type: String,
            default: 'Click on the button to select a Extra Screen',
        }, 
        _class: {
            type: String, 
            default: 'col-md-12',
        },
        label: {
            type: String,
            default: 'Extra Screen',
        },
        singular: {
            type: String,
            default: 'Extra Screen',
        },
        plural: {
            type: String,
            default: 'Extra Screens',
        },
        screenType: {
            type: String,
            default: '',
        },
        formRef: String,
        formControls: {
            type: Boolean,
            default: true,
        },
        readOnly: {
            type: Boolean,
            default: false,
        },
        popOver: { type: String, default:'' },
        popOverContainer: { type: String, default:'' },
        error: {
            type: Array,
            default: function () {
                return [null]
            },
        },
    },

    data() {
        return {
            table: null,
            record: null,
            disabled: null,
            whereRaw: '',
        };
    },

    created() {

        this.whereRaw = this.screenType ? "screentype = '" + this.screenType + "'": '';

        if ( this.screenType == 'M' ) {
            this.debounceFindExtraScreens = debounce((query) => { this.$root.getMatterExtraScreens(query) }, 500);
        } else {
            this.debounceFindExtraScreens = debounce((query) => { this.$root.getExtraScreens(query) }, 500);
        }
    }, 

    mounted () {
        this.$parent.selectExtraScreen = this;
    },    

    methods: {

        setLoadingIndicator() {

            if ( this.screenType == 'M' ) {
                return this.$root.matterExtraScreenDropdownIsLoading;
            } else {
                return this.$root.extraScreenDropdownIsLoading;
            }

        },

        setDropDownOptions() {

            if ( this.screenType == 'M' ) {
                return this.$root.matterExtraScreens;
            } else {
                return this.$root.extraScreens;
            }

        },

        clearIconClicked() {
            if ( typeof this.$parent[this.clearIconCallback] === 'function') this.$parent[this.clearIconCallback]();
        },

        newRecordButtonClicked() {

            this.$refs['select-extra-screen-form'].table = this.$refs['select-extra-screen-table'];
            this.$refs['select-extra-screen-form'].insertClicked();

        },

        selectExtraScreen() {

            this.show();

            this.table.selectTableFlag = true;

            this.table.whereRaw = this.whereRaw ? [this.whereRaw] : null;

            this.table.loadDataTable();

        },

        extraScreenSelected(selectedRecord) {

            this.record = {
                recordid: selectedRecord.recordid,
                description: selectedRecord.description
            };

            if ( typeof this.$parent.extraScreenSelected === 'function') this.$parent.extraScreenSelected(selectedRecord.recordid, selectedRecord.screentitle);

            this.hide();

        },        

        selectRecord(id) {

            this.extraScreenSelected( this.table.table.row('#' + id).data() );

        },
    },

}  
</script>
